import { db } from '../db';
import { customers, customerEmbeddings, embeddingJobs } from '@shared/schema';
import { eq, and, sql, count, isNull } from 'drizzle-orm';
import { safeApiCall, ExternalApiError } from '../utils/error-handler';
import { applicationLogger } from './application-logger';

// Environment variables with defaults and validation
const OPENAI_API_KEY = process.env.OPENAI_API_KEY;
const EMBEDDING_MODEL = process.env.EMBEDDING_MODEL || 'text-embedding-3-small';
const EMBED_BATCH_SIZE = parseInt(process.env.EMBED_BATCH_SIZE || '100');
const OPENAI_TIMEOUT_MS = parseInt(process.env.OPENAI_TIMEOUT_MS || '30000');
const OPENAI_MAX_RETRIES = parseInt(process.env.OPENAI_MAX_RETRIES || '3');
const RATE_LIMIT_MAX_RETRIES = parseInt(process.env.RATE_LIMIT_MAX_RETRIES || '5');

// Cancellation optimization constants
const CANCELLATION_TIMEOUT_MS = parseInt(process.env.CANCELLATION_TIMEOUT_MS || '5000');
const CANCEL_RATE_LIMIT_WINDOW_MS = parseInt(process.env.CANCEL_RATE_LIMIT_WINDOW_MS || '10000'); // 10 seconds
const MAX_CANCEL_REQUESTS_PER_WINDOW = parseInt(process.env.MAX_CANCEL_REQUESTS_PER_WINDOW || '5'); // Max 5 cancel requests per 10 seconds

export interface JobStatus {
  jobId: string;
  status: 'idle' | 'running' | 'cancelling' | 'cancelled' | 'completed' | 'failed';
  totalCustomers: number;
  processedCustomers: number;
  batchSize: number;
  estimatedTokensSaved?: number;
  error?: string;
  createdAt: Date;
  startedAt?: Date;
  completedAt?: Date;
  cancelledAt?: Date;
  retriedCustomers?: number;
  apiCallsCount?: number;
}

interface EmbeddingApiResponse {
  object: string;
  data: Array<{
    object: string;
    embedding: number[];
    index: number;
  }>;
  model: string;
  usage: {
    prompt_tokens: number;
    total_tokens: number;
  };
}

interface OpenAIErrorResponse {
  error: {
    message: string;
    type: string;
    code?: string;
  };
}

/**
 * Enhanced CancellableEmbeddingService with robust OpenAI API communication
 *
 * Key Enhancements:
 * - Rate limit handling with exponential backoff and jitter
 * - Comprehensive retry logic for transient errors
 * - Enhanced error categorization and handling
 * - Configurable timeouts and retry parameters
 * - Accurate token estimation based on actual text length
 * - Detailed logging and monitoring
 */
export class EnhancedCancellableEmbeddingService {
  private abortControllers: Map<string, AbortController> = new Map();
  private apiCallsCount: Map<string, number> = new Map();
  private retriedCustomersCount: Map<string, number> = new Map();
  
  // Security and performance tracking
  private cancelRequestTimestamps: Map<string, number[]> = new Map(); // Track cancel request timestamps for rate limiting
  private activeJobs: Set<string> = new Set(); // Track active job IDs for quick validation

  /**
   * Start a new cancellable embedding job
   */
  async startEmbeddingJob(): Promise<{ jobId: string }> {
    await applicationLogger.info('vector', 'Starting new embedding job...');

    if (!OPENAI_API_KEY) {
      throw new Error('OPENAI_API_KEY environment variable is required');
    }

    // Get total customers without embeddings
    const customersNeedingEmbeddings = await db
      .select({ count: count() })
      .from(customers)
      .leftJoin(customerEmbeddings, eq(customers.id, customerEmbeddings.customerId))
      .where(isNull(customerEmbeddings.customerId));

    const totalCustomers = customersNeedingEmbeddings[0]?.count || 0;

    if (totalCustomers === 0) {
      throw new Error('No customers need embeddings');
    }

    // Create new job record
    const [job] = await db
      .insert(embeddingJobs)
      .values({
        status: 'idle',
        totalCustomers,
        processedCustomers: 0,
        batchSize: EMBED_BATCH_SIZE,
        cancelRequested: false,
      })
      .returning();

    const jobId = job.id;
    this.apiCallsCount.set(jobId, 0);
    this.retriedCustomersCount.set(jobId, 0);
    this.activeJobs.add(jobId); // Track active job

    // Start background processing
    this.processEmbeddingJob(jobId).catch(error => {
      applicationLogger.error('vector', `Job ${jobId} failed`, error instanceof Error ? error : undefined, { errorMessage: error.message });
      this.updateJobStatus(jobId, 'failed', { error: error.message });
    });

    return { jobId };
  }

  /**
   * Cancel an embedding job with immediate network abortion
   * Optimized for <5 second response time with enhanced security
   */
  async cancelEmbeddingJob(jobId: string): Promise<{ ok: boolean }> {
    // SECURITY: Validate jobId format (UUID)
    if (!this.isValidUUID(jobId)) {
      await applicationLogger.warn('vector', 'Invalid jobId format in cancel request', { jobId });
      throw new Error('Invalid job ID format');
    }

    // SECURITY: Rate limiting for cancel requests
    if (!this.checkCancelRateLimit(jobId)) {
      await applicationLogger.warn('vector', 'Cancel rate limit exceeded', { jobId });
      throw new Error('Too many cancel requests. Please wait before trying again.');
    }

    // DEFENSIVE: Check if job exists and is cancellable
    const jobExists = await this.validateJobExistsAndCancellable(jobId);
    if (!jobExists) {
      await applicationLogger.warn('vector', 'Attempted to cancel non-existent or non-cancellable job', { jobId });
      return { ok: false };
    }

    await applicationLogger.info('vector', `[IMMEDIATE ABORT] Cancelling job ${jobId}`);

    try {
      // OPTIMIZATION: Immediate network abortion - abort ALL in-flight requests first
      const abortController = this.abortControllers.get(jobId);
      if (abortController && !abortController.signal.aborted) {
        abortController.abort();
        await applicationLogger.info('vector', `[IMMEDIATE ABORT] Network requests aborted for job ${jobId}`);
      }

      // OPTIMIZATION: Single atomic database update - skip 'cancelling' state, go directly to 'cancelled'
      const estimatedTokensSaved = await this.calculateTokensSavedForJob(jobId);
      await this.updateJobStatus(jobId, 'cancelled', {
        cancelRequested: true,
        estimatedTokensSaved,
        cancelledAt: new Date()
      });

      // OPTIMIZATION: Immediate resource cleanup (non-blocking)
      setImmediate(() => this.cleanupJobResources(jobId));

      await applicationLogger.info('vector', `[IMMEDIATE ABORT] Job ${jobId} cancelled successfully in <1s`);
      return { ok: true };

    } catch (error) {
      await applicationLogger.error('vector', `Failed to cancel job ${jobId}`, error instanceof Error ? error : undefined, {
        errorMessage: error instanceof Error ? error.message : 'Unknown error',
        jobId
      });
      
      // DEFENSIVE: Ensure cleanup even on error
      setImmediate(() => this.cleanupJobResources(jobId));
      
      return { ok: false };
    }
  }

  /**
   * Get job status with enhanced metrics
   */
  async getJobStatus(jobId: string): Promise<JobStatus | null> {
    const [job] = await db
      .select()
      .from(embeddingJobs)
      .where(eq(embeddingJobs.id, jobId));

    if (!job) return null;

    return {
      jobId: job.id,
      status: job.status as JobStatus['status'],
      totalCustomers: job.totalCustomers,
      processedCustomers: job.processedCustomers,
      batchSize: job.batchSize,
      estimatedTokensSaved: job.estimatedTokensSaved || undefined,
      error: job.errorMessage || undefined,
      createdAt: job.createdAt!,
      startedAt: job.startedAt || undefined,
      completedAt: job.completedAt || undefined,
      cancelledAt: job.cancelledAt || undefined,
      retriedCustomers: this.retriedCustomersCount.get(jobId) || 0,
      apiCallsCount: this.apiCallsCount.get(jobId) || 0,
    };
  }

  /**
   * Get the latest active job status for dashboard
   */
  async getLatestJobStatus(): Promise<JobStatus | null> {
    const [job] = await db
      .select()
      .from(embeddingJobs)
      .orderBy(sql`${embeddingJobs.createdAt} DESC`)
      .limit(1);

    if (!job) return null;

    return this.getJobStatus(job.id);
  }

  /**
   * Background job processing with enhanced error handling
   */
  private async processEmbeddingJob(jobId: string): Promise<void> {
    const abortController = new AbortController();
    this.abortControllers.set(jobId, abortController);

    try {
      // Update job to running
      await this.updateJobStatus(jobId, 'running', { startedAt: new Date() });

      // Get customers needing embeddings
      const customersToProcess = await this.getCustomersNeedingEmbeddings();

      let processedCount = 0;
      let estimatedTokensSaved = 0;

      // Process in batches with optimized cancellation checks
      for (let i = 0; i < customersToProcess.length; i += EMBED_BATCH_SIZE) {
        // OPTIMIZATION: Batch-level cancellation check only (99% reduction in check frequency)
        if (abortController.signal.aborted) {
          estimatedTokensSaved = this.calculateTokensSaved(customersToProcess.length - processedCount);
          await this.updateJobStatus(jobId, 'cancelled', {
            estimatedTokensSaved,
            cancelledAt: new Date()
          });
          await applicationLogger.info('vector', `[BATCH CANCEL] Job ${jobId} cancelled at batch ${Math.floor(i / EMBED_BATCH_SIZE) + 1}`);
          return;
        }

        const batch = customersToProcess.slice(i, i + EMBED_BATCH_SIZE);

        try {
          await this.processBatchWithEnhancedErrorHandling(batch, abortController.signal, jobId);
          processedCount += batch.length;

          // Update progress
          await this.updateJobStatus(jobId, 'running', { processedCustomers: processedCount });

        } catch (error) {
          if (error instanceof Error && error.name === 'AbortError') {
            // Handle abort gracefully
            estimatedTokensSaved = this.calculateTokensSaved(customersToProcess.length - processedCount);
            await this.updateJobStatus(jobId, 'cancelled', {
              estimatedTokensSaved,
              cancelledAt: new Date()
            });
            return;
          }
          throw error;
        }

        // Small delay between batches to be nice to the API
        await new Promise(resolve => setTimeout(resolve, 200));
      }

      // Job completed successfully
      await this.updateJobStatus(jobId, 'completed', { completedAt: new Date() });
      await applicationLogger.info('vector', `Job ${jobId} completed successfully`, {
        customersProcessed: processedCount,
        apiCalls: this.apiCallsCount.get(jobId) || 0
      });

    } catch (error) {
      await applicationLogger.error('vector', `Job ${jobId} failed`, error instanceof Error ? error : undefined, { errorMessage: error instanceof Error ? error.message : 'Unknown error' });
      await this.updateJobStatus(jobId, 'failed', { error: error instanceof Error ? error.message : 'Unknown error' });
    } finally {
      // OPTIMIZATION: Use consolidated cleanup method
      this.cleanupJobResources(jobId);
    }
  }

  /**
   * Process a batch of customers with enhanced error handling and retry logic
   * OPTIMIZATION: Removed per-customer cancellation checks for performance
   */
  private async processBatchWithEnhancedErrorHandling(batch: any[], signal: AbortSignal, jobId: string): Promise<void> {
    // OPTIMIZATION: Single cancellation check at batch start instead of per-customer
    if (signal.aborted) {
      const abortError = new Error('AbortError');
      abortError.name = 'AbortError';
      throw abortError;
    }

    // Process each customer individually for better error isolation
    for (const customer of batch) {

      // Validate customer data structure
      if (!customer?.id) {
        await applicationLogger.warn('vector', 'Invalid customer data', { customer });
        continue;
      }

      try {
        // Generate embedding with enhanced error handling
        const embedding = await this.generateEmbeddingWithEnhancedErrorHandling(customer, signal, jobId);

        // Use Drizzle ORM with upsert for transaction safety
        await db
          .insert(customerEmbeddings)
          .values({
            customerId: customer.id,
            embedding: embedding,
            embeddingType: 'customer_profile',
            lastGeneratedAt: new Date(),
          })
          .onConflictDoUpdate({
            target: customerEmbeddings.customerId,
            set: {
              embedding: embedding,
              lastGeneratedAt: new Date(),
            },
          });

      } catch (error) {
        // Propagate cancellation errors immediately
        if (error instanceof Error && error.name === 'AbortError') {
          throw error;
        }

        await applicationLogger.error('vector', `Failed to process customer ${customer.id}`, error instanceof Error ? error : undefined, { 
          errorMessage: error instanceof Error ? error.message : 'Unknown error',
          customerId: customer.id
        });

        // Continue with other customers instead of failing the whole batch
      }
    }
  }

  /**
   * Generate embedding with enhanced OpenAI API communication
   * Includes rate limit handling, exponential backoff with jitter, and comprehensive error handling
   */
  private async generateEmbeddingWithEnhancedErrorHandling(
    customer: any,
    signal: AbortSignal,
    jobId: string
  ): Promise<number[]> {
    // Validate API key
    if (!OPENAI_API_KEY) {
      throw new Error('OPENAI_API_KEY is required');
    }

    // Create customer profile text with enhanced validation
    const profileText = this.createEnhancedCustomerProfileText(customer);
    
    if (!profileText.trim()) {
      throw new Error(`Customer ${customer.id} has no valid profile data`);
    }

    // Increment API call counter
    const currentCount = this.apiCallsCount.get(jobId) || 0;
    this.apiCallsCount.set(jobId, currentCount + 1);

    // Use safeApiCall with enhanced retry logic for OpenAI-specific errors
    const result = await this.callOpenAIWithRetry(profileText, signal, jobId);

    if (!result.success) {
      throw new ExternalApiError(result.error || 'OpenAI API call failed');
    }

    return result.data!;
  }

  /**
   * Call OpenAI API with comprehensive retry logic and error handling
   */
  private async callOpenAIWithRetry(
    profileText: string,
    signal: AbortSignal,
    jobId: string
  ): Promise<{ success: boolean; data?: number[]; error?: string }> {
    let lastError: Error | null = null;
    let attempt = 0;
    const maxRetries = RATE_LIMIT_MAX_RETRIES;

    while (attempt < maxRetries) {
      attempt++;

      try {
        // Check for cancellation before each attempt
        if (signal.aborted) {
          const abortError = new Error('AbortError');
          abortError.name = 'AbortError';
          throw abortError;
        }

        const embedding = await this.makeOpenAIRequest(profileText, signal);
        return { success: true, data: embedding };

      } catch (error) {
        lastError = error as Error;

        // Don't retry on cancellation
        if (error instanceof Error && error.name === 'AbortError') {
          throw error;
        }

        // Parse error response for better handling
        const errorInfo = this.parseOpenAIError(error);

        // Don't retry on client errors (400-499) except rate limits
        if (errorInfo.isClientError && !errorInfo.isRateLimit) {
          await applicationLogger.error('vector', 'OpenAI client error - not retrying', undefined, {
            errorMessage: errorInfo.message,
            statusCode: errorInfo.statusCode,
            errorType: errorInfo.type
          });
          break;
        }

        // Don't retry on authentication errors
        if (errorInfo.isAuthError) {
          await applicationLogger.error('vector', 'OpenAI authentication error - not retrying', undefined, {
            errorMessage: errorInfo.message
          });
          break;
        }

        // Log retry attempt
        await applicationLogger.warn('vector', `OpenAI API call failed, attempt ${attempt}/${maxRetries}`, {
          error: errorInfo.message,
          statusCode: errorInfo.statusCode,
          isRateLimit: errorInfo.isRateLimit,
          jobId
        });

        // Don't retry if we've hit max attempts
        if (attempt >= maxRetries) {
          break;
        }

        // Calculate delay with exponential backoff and jitter
        const delay = this.calculateRetryDelay(attempt, errorInfo.isRateLimit);
        
        await applicationLogger.info('vector', `Retrying OpenAI API call in ${delay}ms`, {
          attempt,
          delay,
          isRateLimit: errorInfo.isRateLimit
        });

        // Wait with timeout check
        await this.delayWithCancellation(delay, signal);

        // Increment retry counter
        const retryCount = this.retriedCustomersCount.get(jobId) || 0;
        this.retriedCustomersCount.set(jobId, retryCount + 1);
      }
    }

    return {
      success: false,
      error: lastError?.message || 'OpenAI API call failed after retries'
    };
  }

  /**
   * Make the actual OpenAI API request with proper formatting and timeout
   */
  private async makeOpenAIRequest(profileText: string, signal: AbortSignal): Promise<number[]> {
    // Create timeout signal
    const timeoutController = new AbortController();
    const timeoutId = setTimeout(() => timeoutController.abort(), OPENAI_TIMEOUT_MS);

    // Combine abort signals
    const combinedSignal = this.combineAbortSignals([signal, timeoutController.signal]);

    try {
      const response = await fetch('https://api.openai.com/v1/embeddings', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${OPENAI_API_KEY}`,
          'Content-Type': 'application/json',
          'User-Agent': 'Smart-CDP/1.0.0'
        },
        body: JSON.stringify({
          input: profileText,
          model: EMBEDDING_MODEL,
          encoding_format: 'float'
        }),
        signal: combinedSignal,
      });

      clearTimeout(timeoutId);

      // Handle non-OK responses
      if (!response.ok) {
        let errorDetails = `${response.status} ${response.statusText}`;
        
        try {
          const errorBody: OpenAIErrorResponse = await response.json();
          if (errorBody.error?.message) {
            errorDetails = errorBody.error.message;
          }
        } catch {
          // Failed to parse error response, use status text
        }

        const error = new ExternalApiError(`OpenAI API error: ${errorDetails}`, response.status);
        (error as any).response = response;
        throw error;
      }

      // Parse successful response
      const data: EmbeddingApiResponse = await response.json();
      
      // Validate response structure
      if (!data.data || !Array.isArray(data.data) || data.data.length === 0) {
        throw new Error('Invalid response structure from OpenAI API');
      }

      const embedding = data.data[0].embedding;

      // Validate embedding format
      if (!Array.isArray(embedding) || embedding.length === 0) {
        throw new Error(`Invalid embedding format received from OpenAI: ${typeof embedding}`);
      }

      // Ensure all values are numbers
      if (!embedding.every(val => typeof val === 'number' && !isNaN(val))) {
        throw new Error('Embedding contains invalid numeric values');
      }

      // Convert to Float32 array and back to regular array for consistency
      const floatArray = new Float32Array(embedding);
      return Array.from(floatArray);

    } catch (error) {
      clearTimeout(timeoutId);
      
      // Handle timeout specifically
      if (error instanceof Error && error.name === 'AbortError') {
        if (timeoutController.signal.aborted) {
          throw new Error(`OpenAI API request timed out after ${OPENAI_TIMEOUT_MS}ms`);
        }
      }
      
      throw error;
    }
  }

  /**
   * Create enhanced customer profile text with better validation
   */
  private createEnhancedCustomerProfileText(customer: any): string {
    const parts: string[] = [];

    // Helper function to safely add field
    const addField = (label: string, value: any) => {
      if (value !== null && value !== undefined && value !== '') {
        const stringValue = typeof value === 'string' ? value : String(value);
        if (stringValue.trim()) {
          parts.push(`${label}: ${stringValue.trim()}`);
        }
      }
    };

    // Add customer fields with validation
    addField('Name', [customer.firstName, customer.lastName].filter(Boolean).join(' '));
    addField('Email', customer.email);
    addField('Segment', customer.customerSegment);
    addField('LTV', customer.lifetimeValue ? `$${customer.lifetimeValue}` : null);
    
    if (customer.currentAddress) {
      const addr = typeof customer.currentAddress === 'string'
        ? customer.currentAddress
        : JSON.stringify(customer.currentAddress);
      addField('Address', addr);
    }

    return parts.join(', ');
  }

  /**
   * Parse OpenAI API errors for better error handling decisions
   */
  private parseOpenAIError(error: any): {
    message: string;
    statusCode?: number;
    type?: string;
    isRateLimit: boolean;
    isClientError: boolean;
    isAuthError: boolean;
  } {
    let message = error.message || 'Unknown error';
    let statusCode: number | undefined;
    let type: string | undefined;

    // Extract status code from ExternalApiError
    if (error instanceof ExternalApiError) {
      statusCode = error.statusCode;
    }

    // Extract status code from error message
    const statusMatch = message.match(/(\d{3})/);
    if (statusMatch && !statusCode) {
      statusCode = parseInt(statusMatch[1]);
    }

    // Determine error characteristics
    const isRateLimit = statusCode === 429 || message.toLowerCase().includes('rate limit');
    const isClientError = statusCode ? statusCode >= 400 && statusCode < 500 : false;
    const isAuthError = statusCode === 401 || statusCode === 403 || 
                       message.toLowerCase().includes('unauthorized') ||
                       message.toLowerCase().includes('forbidden') ||
                       message.toLowerCase().includes('api key');

    return {
      message,
      statusCode,
      type,
      isRateLimit,
      isClientError,
      isAuthError
    };
  }

  /**
   * Calculate retry delay with exponential backoff and jitter
   */
  private calculateRetryDelay(attempt: number, isRateLimit: boolean): number {
    // Base delay: higher for rate limits
    const baseDelay = isRateLimit ? 5000 : 1000; // 5s for rate limits, 1s for others
    
    // Exponential backoff
    const exponentialDelay = baseDelay * Math.pow(2, attempt - 1);
    
    // Cap the delay (max 60s for rate limits, 30s for others)
    const maxDelay = isRateLimit ? 60000 : 30000;
    const cappedDelay = Math.min(exponentialDelay, maxDelay);
    
    // Add jitter (±25% of the delay)
    const jitterRange = cappedDelay * 0.25;
    const jitter = (Math.random() - 0.5) * 2 * jitterRange;
    
    return Math.max(1000, Math.floor(cappedDelay + jitter)); // Minimum 1s delay
  }

  /**
   * Delay with cancellation support
   */
  private async delayWithCancellation(delayMs: number, signal: AbortSignal): Promise<void> {
    return new Promise((resolve, reject) => {
      const timeoutId = setTimeout(resolve, delayMs);
      
      const abortHandler = () => {
        clearTimeout(timeoutId);
        const abortError = new Error('AbortError');
        abortError.name = 'AbortError';
        reject(abortError);
      };

      if (signal.aborted) {
        clearTimeout(timeoutId);
        abortHandler();
        return;
      }

      signal.addEventListener('abort', abortHandler, { once: true });
    });
  }

  /**
   * Combine multiple abort signals
   */
  private combineAbortSignals(signals: AbortSignal[]): AbortSignal {
    const controller = new AbortController();

    for (const signal of signals) {
      if (signal.aborted) {
        controller.abort();
        break;
      }
      signal.addEventListener('abort', () => controller.abort(), { once: true });
    }

    return controller.signal;
  }

  /**
   * Get customers that need embeddings
   */
  private async getCustomersNeedingEmbeddings(): Promise<any[]> {
    const results = await db
      .select({
        id: customers.id,
        firstName: customers.firstName,
        lastName: customers.lastName,
        email: customers.email,
        customerSegment: customers.customerSegment,
        lifetimeValue: customers.lifetimeValue,
        currentAddress: customers.currentAddress,
      })
      .from(customers)
      .leftJoin(customerEmbeddings, eq(customers.id, customerEmbeddings.customerId))
      .where(isNull(customerEmbeddings.customerId))
      .limit(10000); // Safety limit

    return results;
  }

  /**
   * SECURITY: Validate jobId is a valid UUID format
   */
  private isValidUUID(jobId: string): boolean {
    const uuidRegex = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i;
    return typeof jobId === 'string' && uuidRegex.test(jobId);
  }

  /**
   * SECURITY: Rate limiting for cancel requests to prevent abuse
   */
  private checkCancelRateLimit(jobId: string): boolean {
    const now = Date.now();
    const timestamps = this.cancelRequestTimestamps.get(jobId) || [];
    
    // Remove timestamps older than the window
    const validTimestamps = timestamps.filter(ts => now - ts < CANCEL_RATE_LIMIT_WINDOW_MS);
    
    // Check if we've exceeded the limit
    if (validTimestamps.length >= MAX_CANCEL_REQUESTS_PER_WINDOW) {
      return false;
    }
    
    // Add current timestamp and update
    validTimestamps.push(now);
    this.cancelRequestTimestamps.set(jobId, validTimestamps);
    
    return true;
  }

  /**
   * DEFENSIVE: Validate job exists and is in a cancellable state
   */
  private async validateJobExistsAndCancellable(jobId: string): Promise<boolean> {
    try {
      const [job] = await db
        .select({ status: embeddingJobs.status })
        .from(embeddingJobs)
        .where(eq(embeddingJobs.id, jobId));

      if (!job) {
        return false;
      }

      // Only allow cancellation for running jobs
      const cancellableStates = ['idle', 'running'];
      return cancellableStates.includes(job.status);
      
    } catch (error) {
      await applicationLogger.error('vector', 'Error validating job for cancellation', error instanceof Error ? error : undefined, { jobId });
      return false;
    }
  }

  /**
   * OPTIMIZATION: Calculate tokens saved for a specific job
   */
  private async calculateTokensSavedForJob(jobId: string): Promise<number> {
    try {
      const [job] = await db
        .select({
          totalCustomers: embeddingJobs.totalCustomers,
          processedCustomers: embeddingJobs.processedCustomers
        })
        .from(embeddingJobs)
        .where(eq(embeddingJobs.id, jobId));

      if (!job) {
        return 0;
      }

      const remainingCustomers = job.totalCustomers - job.processedCustomers;
      return this.calculateTokensSaved(remainingCustomers);
      
    } catch (error) {
      await applicationLogger.warn('vector', 'Error calculating tokens saved', { jobId });
      return 0;
    }
  }

  /**
   * OPTIMIZATION: Consolidated resource cleanup method
   * Handles all job-related resources in a single atomic operation
   */
  private cleanupJobResources(jobId: string): void {
    try {
      // Clean up AbortController
      const abortController = this.abortControllers.get(jobId);
      if (abortController && !abortController.signal.aborted) {
        abortController.abort();
      }
      this.abortControllers.delete(jobId);
      
      // Clean up tracking maps
      this.apiCallsCount.delete(jobId);
      this.retriedCustomersCount.delete(jobId);
      this.activeJobs.delete(jobId);
      
      // Clean up rate limiting data (keep recent entries for security)
      const now = Date.now();
      const timestamps = this.cancelRequestTimestamps.get(jobId) || [];
      const recentTimestamps = timestamps.filter(ts => now - ts < CANCEL_RATE_LIMIT_WINDOW_MS);
      
      if (recentTimestamps.length > 0) {
        this.cancelRequestTimestamps.set(jobId, recentTimestamps);
      } else {
        this.cancelRequestTimestamps.delete(jobId);
      }
      
    } catch (error) {
      // Silent cleanup failure - don't propagate errors
      console.warn(`[CLEANUP] Failed to clean resources for job ${jobId}:`, error);
    }
  }

  /**
   * Update job status and related fields with timeout protection
   */
  private async updateJobStatus(
    jobId: string,
    status: JobStatus['status'],
    updates: Partial<{
      processedCustomers: number;
      cancelRequested: boolean;
      estimatedTokensSaved: number;
      error: string;
      startedAt: Date;
      completedAt: Date;
      cancelledAt: Date;
    }> = {}
  ): Promise<void> {
    // DEFENSIVE: Validate jobId before database operation
    if (!this.isValidUUID(jobId)) {
      throw new Error('Invalid jobId for status update');
    }

    const updateData: any = { status };

    if (updates.processedCustomers !== undefined) updateData.processedCustomers = updates.processedCustomers;
    if (updates.cancelRequested !== undefined) updateData.cancelRequested = updates.cancelRequested;
    if (updates.estimatedTokensSaved !== undefined) updateData.estimatedTokensSaved = updates.estimatedTokensSaved;
    if (updates.error !== undefined) updateData.errorMessage = updates.error;
    if (updates.startedAt !== undefined) updateData.startedAt = updates.startedAt;
    if (updates.completedAt !== undefined) updateData.completedAt = updates.completedAt;
    if (updates.cancelledAt !== undefined) updateData.cancelledAt = updates.cancelledAt;

    // OPTIMIZATION: Add timeout protection for database operations during cancellation
    const timeoutPromise = new Promise((_, reject) => {
      setTimeout(() => reject(new Error('Database update timeout')), CANCELLATION_TIMEOUT_MS);
    });

    try {
      await Promise.race([
        db.update(embeddingJobs).set(updateData).where(eq(embeddingJobs.id, jobId)),
        timeoutPromise
      ]);
    } catch (error) {
      await applicationLogger.error('vector', 'Database update failed or timed out', error instanceof Error ? error : undefined, { jobId, status });
      throw error;
    }
  }

  /**
   * Calculate estimated tokens saved with improved accuracy
   */
  private calculateTokensSaved(remainingCustomers: number): number {
    // More accurate token estimation based on average profile length
    // OpenAI's tokenizer roughly: 1 token ≈ 4 characters for English text
    const avgProfileLength = 120; // Average customer profile text length
    const tokensPerCustomer = Math.ceil(avgProfileLength / 4);
    return remainingCustomers * tokensPerCustomer;
  }
